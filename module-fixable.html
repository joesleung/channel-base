var Fixable = seajs.require('fixable');
var fixable = new Fixable({
    fixableElement: '#fixable',
    hasFixedStyle: true,
    zIndex: 100,
    delay: 50,
    onResizeCallback: function(winWidth){
        if( winWidth < 1370 ){
            $('#fixable').addClass('mini'); 
        }else{
            $('#fixable').removeClass('mini');
        } 
    }
});
